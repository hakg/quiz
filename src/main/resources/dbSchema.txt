# USER 테이블 생성 스키마
CREATE TABLE `USER` (
  `email` varchar(100) NOT NULL,
  `userName` varchar(100) NOT NULL,
  `password` varchar(100) NOT NULL,
  `userGubun` varchar(50) NOT NULL,
  `lastLoginDate` datetime DEFAULT NULL,
  `passwordUpdateDate` datetime DEFAULT NULL,
  PRIMARY KEY (`email`)
);
